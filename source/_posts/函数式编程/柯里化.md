    function curry(fn) {
        let count = fn.length // 获取当前函数期望参数个数
        const params = []
        return (...args) => { // 若当前参数个数 < 期望参数则返回下一个需要参数的函数，知道参数满足，返回当前函数
            params = [...params, ...args]
            if (params.length > count) {
                return fn.call(this, ...params)
            }
        }
    }
